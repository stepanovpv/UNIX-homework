# Создать пользователя, обладающего возможностью выполнять действия от имени суперпользователя.

# При создании пользователя mega указываем, чтобы он входил в группу sudo, позволяющую выполнение от 
# имени суперпользователя
 
pavel@ubuntu:~$ sudo useradd -m -G sudo -s /bin/bash mega

# Добавляем пароль

pavel@ubuntu:~$ sudo passwd mega

# Теперь, переключившишь на пользователя mega, можно выполнять команды от имени суперпользователя.
